Class ArmoredHelmet : BasicArmorPickup
{
	Default
	{
		-INVENTORY.ALWAYSPICKUP
		Radius 20;
		Height 16;
		Inventory.PickupMessage "Armored Helmet";
		Inventory.Icon "HELMA0";
		Armor.SavePercent 66.665;
		Armor.SaveAmount 25;
		Armor.MaxFullAbsorb 5;
	}
	Override Bool Used(Actor User)
	{
		If(User && User.Player && Amount > 0)
		{
			User.A_GiveInventory("ArmoredHelmet_2");
			Destroy();
		}
		Return Super.Used(User);
	}
	States
	{
	Spawn:
		HELM A -1;
		Stop;
	}
}
