Class Knife : AxisWeapon
{
	Default
	{
		Weapon.SelectionOrder 3700;
		+WEAPON.MELEEWEAPON
		+INVENTORY.UNDROPPABLE
		Inventory.PickupMessage "You got the Knife!";
		Tag "Combat Knife";
	}
	States
	{
	Spawn:
		WP0P A -1;
		Stop;
	Select:
		WP0G A 1 A_Raise(30);
		Loop;
	Ready:
		WP0G A 1 A_WeaponReady(WRF_ALLOWRELOAD);
		Loop;
	Deselect:
		WP0G A 1 A_Lower(30);
		Loop;
		
	Fire:
		WP0G BCD 2;
		#### E 2 A_Punch;
		#### F 2;
		Goto Ready;
	}
}
